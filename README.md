## Usage

```typescript
import { isNumber } from "https://raw.githubusercontent.com/ajimae/isNumber/master/mod.ts";

var numberOne = 5;
const checkOne = isNumber(numberOne);
console.log(checkOne); // true

var numberTwo = '5';
const checkTwo = isNumber(numberTwo);
console.log(checkTwo); // false
```
